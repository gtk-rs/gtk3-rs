use crate::translate::*;

bitflags::bitflags! {
    /// Through the [`crate::ParamFlags`] flag values, certain aspects of parameters
    /// can be configured. See also `G_PARAM_STATIC_STRINGS`.
    pub struct ParamFlags: u32 {
        /// the parameter is readable
        const READABLE = 1;
        /// the parameter is writable
        const WRITABLE = 2;
        /// alias for [`Self::Readable`] | [`Self::Writable`]
        const READWRITE = 3;
        /// the parameter will be set upon object construction
        const CONSTRUCT = 4;
        /// the parameter can only be set upon object construction
        const CONSTRUCT_ONLY = 8;
        /// upon parameter conversion (see `g_param_value_convert()`)
        ///  strict validation is not required
        const LAX_VALIDATION = 16;
        const USER_0 = 128;
        const USER_1 = 256;
        const USER_2 = 1024;
        const USER_3 = 2048;
        const USER_4 = 4096;
        const USER_5 = 8192;
        const USER_6 = 16384;
        const USER_7 = 32768;
        const USER_8 = 65536;
        /// calls to [`crate::prelude::ObjectExt::set_property()`] for this
        ///  property will not automatically result in a "notify" signal being
        ///  emitted: the implementation must call [`crate::prelude::ObjectExt::notify()`] themselves
        ///  in case the property actually changes. Since: 2.42.
        const EXPLICIT_NOTIFY = 1073741824;
        /// the parameter is deprecated and will be removed
        ///  in a future version. A warning will be generated if it is used
        ///  while running with G_ENABLE_DIAGNOSTIC=1.
        ///  Since 2.26
        const DEPRECATED = 2147483648;
    }
}

#[doc(hidden)]
impl IntoGlib for ParamFlags {
    type GlibType = gobject_ffi::GParamFlags;

    fn into_glib(self) -> gobject_ffi::GParamFlags {
        self.bits()
    }
}

#[doc(hidden)]
impl FromGlib<gobject_ffi::GParamFlags> for ParamFlags {
    unsafe fn from_glib(value: gobject_ffi::GParamFlags) -> Self {
        Self::from_bits_truncate(value)
    }
}
